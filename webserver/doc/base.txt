CREATE TABLE Types(
   ID_Type INT AUTO_INCREMENT,
   Type VARCHAR(50),
   PRIMARY KEY(ID_Type)
);

CREATE TABLE Actives(
   ID_Active INT AUTO_INCREMENT,
   Active VARCHAR(50),
   PRIMARY KEY(ID_Active)
);

CREATE TABLE Users(
   ID_User INT AUTO_INCREMENT,
   Username VARCHAR(50),
   Password VARCHAR(50),
   Date_Creation DATE,
   email VARCHAR(50),
   prenom VARCHAR(50),
   nom VARCHAR(50),
   ID_Active INT NOT NULL,
   ID_Type INT NOT NULL,
   PRIMARY KEY(ID_User),
   FOREIGN KEY(ID_Active) REFERENCES Actives(ID_Active),
   FOREIGN KEY(ID_Type) REFERENCES Types(ID_Type)
);

CREATE TABLE Comments(
   ID INT AUTO_INCREMENT,
   Title VARCHAR(620),
   Message VARCHAR(4000),
   ID_User INT NOT NULL,
   PRIMARY KEY(ID),
   FOREIGN KEY(ID_User) REFERENCES Users(ID_User)
);
